let editandoId = null; // ===================== // ENUMS // ===================== async function carregarEnums() { let res = await fetch("/api/enums"); let data = await res.json(); let m = document.getElementById("modalidade"); let e = document.getElementById("eixo"); m.innerHTML = ""; e.innerHTML = ""; data.modalidade.forEach(x => m.add(new Option(x, x))); data.eixo.forEach(x => e.add(new Option(x, x))); } // ===================== // LISTAR // ===================== async function carregarTabela() { let res = await fetch("/api/formacoes"); let data = await res.json(); let html = ""; data.forEach(f => { html += ` ${f.id} ${f.descricao} ${f.data_termino || ""} ${f.modalidade || ""} `; }); document.getElementById("tabela").innerHTML = html; } // ===================== // SALVAR / ATUALIZAR // ===================== async function salvar() { let dados = { descricao: document.getElementById("descricao").value.trim(), data_termino: document.getElementById("data_termino").value, carga_horaria: parseInt(document.getElementById("carga").value), modalidade: document.getElementById("modalidade").value, eixo: document.getElementById("eixo").value }; let url = "/api/formacao"; let method = "POST"; if (editandoId) { url = `/api/formacao/${editandoId}`; method = "PUT"; } let res = await fetch(url, { method: method, headers: {"Content-Type": "application/json"}, body: JSON.stringify(dados) }); let r = await res.json(); mostrarMsg(r); if (r.ok) { limpar(); carregarTabela(); } } // ===================== // EDITAR // ===================== function editar(f) { editandoId = f.id; document.getElementById("descricao").value = f.descricao; document.getElementById("data_termino").value = f.data_termino; document.getElementById("carga").value = f.carga_horaria; document.getElementById("modalidade").value = f.modalidade; document.getElementById("eixo").value = f.eixo; } // ===================== // DELETAR // ===================== async function deletar(id) { if (!confirm("Excluir registro?")) return; let res = await fetch(`/api/formacao/${id}`, {method: "DELETE"}); let r = await res.json(); mostrarMsg(r); if (r.ok) carregarTabela(); } // ===================== // UI // ===================== function mostrarMsg(r) { let msg = document.getElementById("msg"); if (r.erro) { msg.innerHTML = `
${r.erro}
`; } else { msg.innerHTML = `
Sucesso
`; } } function limpar() { editandoId = null; document.querySelectorAll("input, select").forEach(el => el.value = ""); } // ===================== // INIT // ===================== carregarEnums(); carregarTabela();
